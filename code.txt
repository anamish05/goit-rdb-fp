CREATE TABLE countries (
country_code varchar(10),
country_name varchar(100)
)

INSERT INTO countries (country_code,country_name)
WITH numbered_regions AS (
    SELECT 
        Entity, 
        -- Creates 01, 02, 03... based on alphabetical order of the Entity name
        LPAD(ROW_NUMBER() OVER (ORDER BY Entity), 2, '0') as region_num
    FROM (SELECT DISTINCT Entity FROM infections WHERE Code IS NULL OR Code = '') as regions
)
-- Part 1: Insert real countries that already have a Code
SELECT DISTINCT Code, Entity 
FROM infections 
WHERE Code IS NOT NULL AND Code != ''

UNION ALL

-- Part 2: Insert regions with your new R01, R02 format
SELECT 
    CONCAT('R', region_num), 
    Entity 
FROM numbered_regions;

CREATE TABLE infectious_cases AS
SELECT 
    c.country_code AS Code, -- pull created codes
    i.Year, 
    i.Number_yaws, 
    i.polio_cases, 
    i.cases_guinea_worm, 
    i.Number_rabies, 
    i.Number_malaria, 
    i.Number_hiv, 
    i.Number_tuberculosis, 
    i.Number_smallpox, 
    i.Number_cholera_cases
FROM infections i
JOIN countries c ON i.Entity = c.country_name;


---Task3------------

select Code, min(Number_rabies) as min_rub, max(Number_rabies) as max_rub, 
avg(Number_rabies) as avg_rub, sum(Number_rabies) as sum_rub
from infectious_cases
group by Code
having min_rub is not null
order by avg_rub desc
limit 10;

DELIMITER //

CREATE FUNCTION Create_diff(input_year INT)
RETURNS INT
DETERMINISTIC 
NO SQL
BEGIN
    DECLARE result INT;
    SET result = timestampdiff(year,makedate(input_year,1),curdate());
    RETURN result;
END //

DELIMITER ;


select Code, year, Create_diff(year) as diff_year
from infectious_cases;
